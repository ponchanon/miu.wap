Exercise 02

Write a function printNumbers(from, to) that outputs a number every second, 
            starting from from and ending with to

Solution:

function printNumbers(from, to) {
    let start = from;
    console.log(`Printing from ${from} to ${to}`);
    const intervalId = setInterval(() => {
        console.log(start++);
        if (start > to) {
            //start=from; //this will help for continuous run 
            clearInterval(intervalId);
        }
    }, 1000);
}
// Calling the function to test the program
printNumbers(25, 33);
