        function reverse(str){
            let x = str.length-1;
            let reversed = '';
            while(x>=0){
                reversed += str.charAt(x);
                x--;
            }
            return reversed;
        }

function findSecondBiggest(a){
    let max = a[0];
    let second = 0;    
    for (var i = 0, n = a.length; i < n; ++i) {
        var nr = +a[i];    
        if (nr > max) {
            second = max; 
            max = nr;
        } else if (nr < max && nr > second) {
            second = nr; 
        }
    }
    return second;
}

function printFibo(n,a,b){
    let arr = [a, b];    
    if(n==1) return arr = [a];
    
    for (let i = 2; i < n; i++){
        arr.push(arr[i - 2] +  arr[i -1]);        
    }
    return arr.join(',');
}
